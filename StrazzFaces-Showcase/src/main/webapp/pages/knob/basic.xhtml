Knob

s:knob it's a wrapper of this jQuery plugin by Anthony Terrien. It's useful to insert numeric values in a range. 


<p:growl autoUpdate="true"/>
<p:outputPanel layout="block" styleClass="container" id="container">
	<div class="row">
		<div class="col-md-2">
		</div>
		<div class="col-md-8 text-center">
			<s:knob value="#{'#{knobBasicBean.value}'}">
				<p:ajax process="@this" listener="#{'#{knobBasicBean.confirm}'}" />
			</s:knob>
		</div>
		<div class="col-md-2">
		</div>
	</div>
</p:outputPanel>
				

@ManagedBean
@RequestScoped
public class KnobBasicBean implements Serializable {

	private static final long serialVersionUID = 1L;

	private int value = 25;

	public void confirm() {
		FacesContext.getCurrentInstance().addMessage(null, new FacesMessage("You chose " + value));
	}

	public int getValue() {
		return value;
	}

	public void setValue(int value) {
		this.value = value;
	}
}
