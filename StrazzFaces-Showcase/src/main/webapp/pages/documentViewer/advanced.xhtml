	<p:spinner size="10" value="#{advancedDocumentViewerBean.page}" min="1" max="69">
		<p:ajax process="@this" update="@parent"/>

	<s:documentViewer
		height="800"
		unmapped="true"
		locale="af"
		page="#{'#{advancedDocumentViewerBean.page}'}"
		library="books"
		name="AliceInWonderland.pdf" />
</p:outputPanel>
					
				
					
@ManagedBean
@SessionScoped
public class AdvancedDocumentViewerBean implements Serializable {
	
	private int page = 2;
	
	public int getPage() {
		return page;
	}

	public void setPage(int page) {
		this.page = page;
	}
}
